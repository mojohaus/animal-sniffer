47      public static void main( String[] args )
48          throws IOException
49      {
50          if ( args.length == 0 )
51          {
52              System.err.println( "Usage: java -jar animal-sniffer.jar [JAR/CLASS FILES]" );
53              System.err.println("  -h   : show a human readable Java version number");
54              System.err.println("  -t N : return exit code 1 if any file has a class file version number > N");
55              System.exit( -1 );
56          }
57  
58          Mainmal_sniffer/Main.html#Main">Main m = new Main();
59          String threshold = null;
60  
61          List<File> files = new ArrayList<>();
62          for ( int i = 0; i < args.length; i++ )
63          {
64              if (args[i].equals("-h"))
65              {
66                  m.humanReadableName = true;
67                  continue;
68              }
69              if (args[i].equals("-t"))
70              {
71                  threshold = args[++i];
72                  for ( Map.Entry<String, String> entry : HUMAN_READABLE_NAME.entrySet() )
73                  {
74                      if ( entry.getValue().equals( threshold ) )
75                      {
76                          threshold = entry.getKey();
77                          break;
78                      }
79                  }
80                  continue;
81              }
82  
83              files.add(new File(args[i]));
84          }
85  
86          for ( File f : files )
87          {
88              m.process( f );
89          }
90  
91          if (threshold!=null && m.maximumVersion.compareTo(threshold)>0)
92              System.exit(1);
93      }
94  
95      protected void process( String name, InputStream image )
96          throws IOException
97      {
98          DataInputStream dis = new DataInputStream( image );
99          byte[] buf = new byte[8];
100         dis.readFully( buf );
101 
102         String v = u2(buf[6], buf[7]) + "." + u2(buf[4], buf[5]);
103         if (maximumVersion.compareTo(v)<0)
104         {
105             maximumVersion = v;
106         }
107 
108         if (humanReadableName)
109         {
110             String hn = HUMAN_READABLE_NAME.get(v);
111             if (hn!=null)
112             {
113                 v = hn;
114             }
115         }
116 
117         System.out.println( v + " " + name );
118     }

120     private static int u2( byte u, byte d )
121     {
122         return ( (int) u ) * 256 + d;
123     }
